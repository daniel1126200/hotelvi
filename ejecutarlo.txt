//repaso de git 
git init iniciar repositorio
git add . para identificar lo modificado
git status para ver lo que se cambio o se agrego
git config --global user.name "" para añadirle tu nombre al proyecto
git config --global user.email "" correo
git config --global --unset credential //borrar credenciales
git commit -m ""se añade comentario para registrar lo nuevo
git push si ya esta creado si no, coger el push del repositorio
